7/*!
8 * \~chinese
9 * \brief 提供对 Eigen 列向量类型的格式化支持。
10 *
11 * 这个格式化器调用 `Scalar` 类型的格式化器来格式化每个分量，
12 * 并在各分量之间添加逗号、空格，在开头和结尾添加括号。由于它直接调用 `Scalar`
13 * 类型的格式化器格式化分量，任何可用于 `Scalar` 类型的格式符也都可用于格式化列向量。
14 * 相应地，`Scalar` 类型必须可以被格式化。
15 *
16 * 格式符：与分量类型的格式符相同，例如 `Vector3f` 可以使用所有 `float`
17 * 类型的格式符，`Vector2i` 可以使用所有 `int` 类型的格式符。
18 *
19 * \tparam Scalar 向量中分量的类型
20 * \tparam n_dim 向量维数
21 */
22template<typename Scalar, int n_dim>
23struct fmt::formatter<Eigen::Matrix<Scalar, n_dim, 1>> : formatter<Scalar>
24{
25 static_assert(n_dim > 0, "Dimension of a vector must be positive");
26 fmt::appender format(const Eigen::Matrix<Scalar, n_dim, 1>& v, format_context& ctx) const
27 {
28 fmt::appender iter = fmt::format_to(ctx.out(), "(");
29 iter = formatter<Scalar>::format(v(0), ctx);
30 for (int i = 1; i < n_dim; ++i) {
31 iter = fmt::format_to(iter, ", ");
32 iter = formatter<Scalar>::format(v(i), ctx);
33 }
34 iter = fmt::format_to(iter, ")");
35 return iter;
36 }
37};

39/*!
40 * \~chinese
41 * \brief 提供对 Eigen 方阵类型的格式化支持。
42 *
43 * 这个格式化器调用 `Scalar` 类型的格式化器来格式化每个元素，
44 * 并在同一行的各元素间添加空格，在行间添加换行符。由于它直接调用 `Scalar`
45 * 类型的格式化器来格式化元素，任何可用于 `Scalar` 类型的格式符也都可用于格式化方阵。
46 * 相应地，`Scalar` 类型必须可以被格式化。
47 *
48 * 格式符：与元素类型的格式符相同，例如 `Matrix4f` 可以使用所有 `float`
49 * 类型的格式符，`Matrix3d` 可以使用所有 `double` 类型的格式符。
50 *
51 * 请注意，方阵的最后一行不会添加换行符，如果使用 `fmt::print` / `std::printf` /
52 * `std::cout` 等工具直接输出时，请自行添加换行。使用 spdlog 的日志输出则不必。
53 *
54 * 当矩阵中的元素位数不一时（例如 10 和 0），默认的格式符 `{}` 并不会对齐它们。
55 * 如果使用者希望将数字右对齐，请直接使用 fmtlib 的右对齐格式，例如：
56 * ```cpp
57 * Eigen::Matrix4f m = 10.0f * Eigen::Matrix4f::Identity();
58 * fmt::print("matrix: {:>5.1f}", m);
59 * ```
60 * 会输出
61 * ```
62 * matrix:
63 * 10.0 0.0 0.0 0.0
64 * 0.0 10.0 0.0 0.0
65 * 0.0 0.0 10.0 0.0
66 * 0.0 0.0 0.0 10.0
67 * ```
68 *
69 * \tparam Scalar 方阵中的元素类型
70 * \tparam n_dim 方阵的维数
71 */
72template<typename Scalar, int n_dim>
73struct fmt::formatter<Eigen::Matrix<Scalar, n_dim, n_dim>> : formatter<Scalar>
74{
75 static_assert(n_dim > 0, "Dimension of a square matrix must be positive");
76 fmt::appender format(const Eigen::Matrix<Scalar, n_dim, n_dim>& m, format_context& ctx) const
77 {
78 fmt::appender iter = fmt::format_to(ctx.out(), "\n");
79 for (int row = 0; row < n_dim - 1; ++row) {
80 for (int column = 0; column < n_dim; ++column) {
81 iter = formatter<Scalar>::format(m(row, column), ctx);
82 iter = fmt::format_to(iter, " ");
83 }
84 iter = fmt::format_to(iter, "\n");
85 }
86 for (int column = 0; column < n_dim; ++column) {
87 iter = formatter<Scalar>::format(m(n_dim - 1, column), ctx);
88 iter = fmt::format_to(iter, " ");
89 }
90 return iter;
91 }
92};
