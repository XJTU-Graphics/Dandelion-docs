94/*!
95 * \ingroup utils
96 * \~chinese
97 * \brief 将旋转的四元数表示形式转换为 XYZ 欧拉角表示形式。
98 *
99 * 由于欧拉角固有的缺陷，这个函数并不能完全消除航向角（或称方向角 yaw）在 \f$\pm 90^\circ\f$
100 * 附近时产生的抖动现象。
101 *
102 * 虽然函数的返回值是 `tuple`，但使用 `tuple` 接受返回值并不方便，建议用结构化绑定
103 * ```cpp
104 * auto [x_euler, y_euler, z_euler] = quaternion_to_xyz_euler(w, x, y, z);
105 * ```
106 * 或 `std::tie` 来接受返回值。
107 */
108template<typename T>
109inline std::tuple<T, T, T> quaternion_to_xyz_euler(T w, T x, T y, T z)
110{
111 const T test = x * z + w * y;
112 constexpr T threshold = static_cast<T>(0.5 - 1e-6);
113 T x_rad, y_rad, z_rad;
114 if (std::abs(test) > threshold) {
115 x_rad = static_cast<T>(0.0);
116 y_rad = sign(test) * pi<T>() / static_cast<T>(2.0);
117 z_rad = sign(test) * static_cast<T>(2.0) * std::atan2(x, w);
118 } else {
119 x_rad = std::atan2(-2.0 * (y * z - w * x), squ(w) - squ(x) - squ(y) + squ(z));
120 y_rad = std::asin(2.0f * (x * z + w * y));
121 z_rad = std::atan2(-2.0 * (x * y - w * z), squ(w) + squ(x) - squ(y) - squ(z));
122 }
123 const T x_euler = degrees(x_rad);
124 const T y_euler = degrees(y_rad);
125 const T z_euler = degrees(z_rad);
126 return std::make_tuple(x_euler, y_euler, z_euler);
127}
